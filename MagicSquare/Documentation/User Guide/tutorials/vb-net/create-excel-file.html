How to create Excel file in VB.NET



'-----------------------------------------------------------------------
' Tutorial 03
'
' This tutorial shows how to create an Excel file (1) that has
' multiple sheets (2) in VB.NET. The created Excel file is empty and the
' next tutorial shows how to add data into sheets.
' ----------------------------------------------------------------------
Click here to see the Excel file

Imports EasyXLS

Module Tutorial03

    Sub Main()

        Console.WriteLine("Tutorial 03" & vbCrLf & "----------" & vbCrLf)

        ' Create an instance of the class that creates Excel files, having two sheets (1)
        Dim xls As New ExcelDocument(2)

        ' Set the sheet names (2)
        xls.easy_getSheetAt(0).setSheetName("First tab")
        xls.easy_getSheetAt(1).setSheetName("Second tab")

        ' Create the Excel file
        Console.WriteLine("Writing file C:\Samples\Tutorial03.xls.")
        xls.easy_WriteXLSFile("C:\\Samples\\Tutorial03.xls")

        ' Confirm the creation of Excel file
        Dim sError As String = xls.easy_getError()
        If (sError.Equals("")) Then
            Console.Write(vbCrLf & "File successfully created. Press Enter to Exit...")
        Else
            Console.Write(vbCrLf & "Error encountered: " & sError & vbCrLf & "Press Enter to Exit...")
        End If

        ' Dispose memory
        xls.Dispose()

        Console.ReadLine()
    End Sub

End Module
