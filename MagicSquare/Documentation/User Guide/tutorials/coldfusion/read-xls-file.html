How to read XLS file in ColdFusion



<!--
=====================================================================
Tutorial 36

This tutorial shows how to read an Excel XLS file in ColdFusion (the
XLS file generated by Tutorial 09 as base template), modify
some data and save it to another XLS file (Tutorial36.xls).
=====================================================================
-->
Click here to see the Excel file

Tutorial 36<br>
----------<br>

<!-- Create an instance of the class that reads Excel files -->
<cfobject type="java" class="EasyXLS.ExcelDocument" name="xls" action="CREATE">

<!-- Read XLS file -->
Reading file C:\Samples\Tutorial09.xls<br><br>

<cfif (xls.easy_LoadXLSFile("C:\Samples\Tutorial09.xls") is True)>

    <!-- Get the table of data for the second worksheet -->
    <cfset xlsSecondTable = xls.easy_getSheet("Second tab").easy_getExcelTable()>

    <!-- Write some data to the second sheet -->
    <cfset xlsSecondTable.easy_getCell("A1").setValue("Data added by Tutorial36")>
    <cfloop from="0" to="4" index="column">
        <cfset xlsSecondTable.easy_getCell(1, evaluate(column)).setValue("Data " & evaluate(column + 1))>
    </cfloop>

    <!-- Export the new XLS file -->
    Writing file C:\Samples\Tutorial36.xls<br>
    <cfset ret = xls.easy_WriteXLSFile("C:\Samples\Tutorial36.xls")>

    <!-- Confirm export of Excel file -->
    <cfset sError = xls.easy_getError()>
    <cfif (sError is "")>
        <cfoutput>
            File successfully created.
        </cfoutput>
    <cfelse>
        <cfoutput>
            Error encountered: #sError#
        </cfoutput>
    </cfif>
<cfelse>
    <cfoutput>
        Error encountered: #xls.easy_getError()#
    </cfoutput>
</cfif>

<!-- Dispose memory -->
<cfset xls.Dispose()>
