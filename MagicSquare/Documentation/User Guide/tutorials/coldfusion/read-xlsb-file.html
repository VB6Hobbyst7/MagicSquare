How to read XLSB file in ColdFusion



<!--
======================================================================
Tutorial 38

This tutorial shows how to read an Excel XLSB file in ColdFusion (the
XLSB file generated by Tutorial 29 as base template), modify
some data and save it to another XLSB file (Tutorial38.xlsb).
======================================================================
-->
Click here to see the Excel file

Tutorial 38<br>
----------<br>

<!-- Create an instance of the object that generates Excel files -->
<cfobject type="java" class="EasyXLS.ExcelDocument" name="xls" action="CREATE">

<!-- Read the file -->
Reading file C:\Samples\Tutorial29.xlsb<br><br>

<!-- Confirm reading -->
<cfif (xls.easy_LoadXLSBFile("C:\Samples\Tutorial29.xlsb") is True)>
    <!-- Get the table of the second worksheet -->
    <cfset xlsTable = xls.easy_getSheetAt(1).easy_getExcelTable()>
    
    <!-- Write some data -->
    <cfset xlsTable.easy_getCell("A1").setValue("Data added by Tutorial38")>
    <cfloop from="0" to="4" index="column">
        <cfset xlsTable.easy_getCell(1, evaluate(column)).setValue("Data " & evaluate(column + 1))>
    </cfloop>

    <!-- Generate the file -->
    Writing file C:\Samples\Tutorial38.xlsb<br>
    <cfset ret = xls.easy_WriteXLSBFile("C:\Samples\Tutorial38.xlsb")>
 
    <!-- Confirm generation -->
    <cfset sError = xls.easy_getError()>
    <cfif (sError is "")>
        <cfoutput>
            File successfully created.
        </cfoutput>
    <cfelse>
        <cfoutput>
            Error encountered: #sError#
        </cfoutput>
    </cfif>

<cfelse>
    <cfoutput>
        Error encountered: #xls.easy_getError()#
    </cfoutput>
</cfif>

<!-- Dispose memory -->
<cfset xls.Dispose()>
