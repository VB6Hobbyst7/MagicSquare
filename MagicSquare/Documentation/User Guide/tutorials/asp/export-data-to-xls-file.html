How to export data to XLS file in ASP classic



<%@ Language=VBScript %>

<!-- #INCLUDE FILE="DataType.inc" -->

<%
'============================================================================
' Tutorial 04
'
' This tutorial shows how to export data to XLS file (1) that has
' multiple sheets (2)in ASP classic. The first sheet is filled with data (3).
' ===========================================================================
Click here to see the Excel file

response.write("Tutorial 04<br>")
response.write("----------<br>")

' Create an instance of the class that exports Excel files (1)
Set xls = Server.CreateObject("EasyXLS.ExcelDocument")

' Create two sheets (2)
xls.easy_addWorksheet_2("First tab")
xls.easy_addWorksheet_2("Second tab")

' Get the table of data for the first worksheet (3)
Set xlsFirstTable = xls.easy_getSheetAt(0).easy_getExcelTable()

' Add data in cells for report header
for column = 0 to 4
    xlsFirstTable.easy_getCell(0,column).setValue("Column " & (column + 1))
    xlsFirstTable.easy_getCell(0,column).setDataType(DATATYPE_STRING)
next
xlsFirstTable.easy_getRowAt(0).setHeight(30)

' Add data in cells for report values
for row = 0 to 99
    for column = 0 to 4
        xlsFirstTable.easy_getCell(row+1,column).setValue("Data " & (row + 1) & ", " & (column + 1))
        xlsFirstTable.easy_getCell(row+1,column).setDataType(DATATYPE_STRING)
    next
next

' Set column widths
xlsFirstTable.setColumnWidth_2 0, 70
xlsFirstTable.setColumnWidth_2 1, 100
xlsFirstTable.setColumnWidth_2 2, 70
xlsFirstTable.setColumnWidth_2 3, 100
xlsFirstTable.setColumnWidth_2 4, 70

' Export the XLS file
    response.write("Writing file: C:\Samples\Tutorial04.xls<br>")
xls.easy_WriteXLSFile("C:\Samples\Tutorial04.xls")

' Confirm export of Excel file
if xls.easy_getError() = "" then
    response.write("File successfully created.")
else
    response.write("Error encountered: " + xls.easy_getError())
end if

' Dispose memory
xls.Dispose
%>
