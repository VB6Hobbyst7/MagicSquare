How to create Excel file in Java



package testexceljava;

import EasyXLS.*;

/*---------------------------------------------------------------------
 | Tutorial 03
 |
 | This tutorial shows how to create an Excel file (1) that has
 | multiple sheets (2) in Java. The created Excel file is empty and the 
 | next tutorial shows how to add data into sheets.
 ----------------------------------------------------------------------*/
Click here to see the Excel file

public class Tutorial03 {

    public static void main(String[] args) {
        try {
            System.out.println("Tutorial 03");
            System.out.println("----------");

            // Create an instance of the class that creates Excel files, having two sheets (1) 
            ExcelDocument xls = new ExcelDocument(2);
      
            // Set the sheet names (2) 
            xls.easy_getSheetAt(0).setSheetName("First tab");
            xls.easy_getSheetAt(1).setSheetName("Second tab");
      
            // Create the Excel file
            System.out.println("Writing file: C:\\Samples\\Tutorial03.xls");
            xls.easy_WriteXLSFile("C:\\Samples\\Tutorial03.xls");

            // Confirm the creation of Excel file
            if (xls.easy_getError().equals(""))
                System.out.println("File successfully created.");
            else
                System.out.println("Error encountered: " + xls.easy_getError());

            // Dispose memory
            xls.Dispose();
        }
        catch (Exception ex) {
            ex.printStackTrace();
        }
    }
}
