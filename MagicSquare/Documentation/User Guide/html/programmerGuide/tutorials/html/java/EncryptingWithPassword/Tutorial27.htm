How to protect Excel file with password in Java



package testexceljava;

import EasyXLS.*;

/*----------------------------------------------------------------
 |Tutorial 27
 |
 |This tutorial shows how to encrypt and set the password required for opening a document.
* ----------------------------------------------------------------- */
Click here to see the Excel file

public class Tutorial27 {

    public static void main(String[] args) {
        try {
            System.out.println("Tutorial 27");
            System.out.println("----------");

            //Create an instance of the object that generates Excel files, having 2 sheets (1)
            ExcelDocument xls = new ExcelDocument(2);
            
            //Set the sheet names (2)
            xls.easy_getSheetAt(0).setSheetName("First tab");
            xls.easy_getSheetAt(1).setSheetName("Second tab");
            
            //Set the password required for opening the document
            xls.easy_getOptions().setPasswordToOpen("password");
            
            //Generate the file
            System.out.println("Writing file: C:\\Samples\\Tutorial27.xls");
            xls.easy_WriteXLSFile("C:\\Samples\\Tutorial27.xls");

            //Confirm generation
            if (xls.easy_getError().equals(""))
                System.out.println("File successfully created.");
            else
                System.out.println("Error encountered: " + xls.easy_getError());

            //Dispose memory
            xls.Dispose();
        }
        catch (Exception ex) {
            ex.printStackTrace();
        }
    }
}
