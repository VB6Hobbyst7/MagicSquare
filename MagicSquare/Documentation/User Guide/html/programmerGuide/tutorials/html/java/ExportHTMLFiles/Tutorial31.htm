How to export data to HTML file in Java



package testexceljava;

import EasyXLS.*;
import EasyXLS.Constants.*;

/*----------------------------------------------------------------
 | Tutorial 31
 | This tutorial shows how to export an HTML file.		  
 -----------------------------------------------------------------*/
Click here to see the result

public class Tutorial31{

    public static void main(String[] args) {
        try {
            System.out.println("Tutorial 31");
            System.out.println("----------");

            //Create an instance of the object that generates Excel files, having 2 sheets
            ExcelDocument xls = new ExcelDocument(2);

            //Set the sheet name
            xls.easy_getSheetAt(0).setSheetName("First tab");

            //Get the table of the first worksheet
            ExcelTable xlsFirstTable = ((ExcelWorksheet)xls.easy_getSheetAt(0)).easy_getExcelTable();

            //Add the cells for header
            for (int column=0; column<5; column++)
            {
                xlsFirstTable.easy_getCell(0,column).setValue("Column " + (column + 1));
                xlsFirstTable.easy_getCell(0,column).setDataType(DataType.STRING);
            }

            //Add the cells for data
            for (int row=0; row<100; row++)
            {
                for (int column=0; column<5; column++)
                {
                    xlsFirstTable.easy_getCell(row+1,column).setValue("Data " + (row + 1) + ", " + (column + 1));
                    xlsFirstTable.easy_getCell(row+1,column).setDataType(DataType.STRING);
                }
            }

            // Apply a predefined format to the cells.
            xlsFirstTable.easy_setRangeAutoFormat("A1:E101", new ExcelAutoFormat(Styles.AUTOFORMAT_EASYXLS1));

            //Generate the file
            System.out.println("Writing file: C:\\Samples\\Tutorial31.htm");
            xls.easy_WriteHTMLFile("C:\\Samples\\Tutorial31.htm", "First tab");

            //Confirm generation
            if (xls.easy_getError().equals(""))
                System.out.println("File successfully created.");
            else
                System.out.println("Error encountered: " + xls.easy_getError());

            //Dispose memory
            xls.Dispose();
        }
        catch (Exception ex) {
            ex.printStackTrace();
        }
    }
}
