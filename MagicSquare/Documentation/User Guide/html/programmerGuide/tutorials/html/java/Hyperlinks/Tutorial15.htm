How to add hyperlinks in Java



package testexceljava;

import EasyXLS.*;
import EasyXLS.Constants.*;

/*----------------------------------------------------------------
 | Tutorial 15
* This tutorial shows how to create a Hyperlink. There are 5
* types oh hyperlinks:
* (1) - to an URL; (1) 
* (2) - to a FILE; (2) 
* (3) - to a UNC; (3) 
* (4) - to a CELL in the same file; (4) 
* (5) - to a name (5) 
* The link can be placed over multiple cells.
*
* Every type of hyperlink accepts a tool tip description.
*
* Every type of hyperlink accepts a text mark. A text mark is a
* link inside the file. Examples:
* http://www.mysite.com/index.htm#Chapter3
* c:\myfile.xls#Sheet2!D3
*
* Note: The blue links are referenced in the code sample below.
* ----------------------------------------------------------------- */
Click here to see the Excel file

public class Tutorial15 {

    public static void main(String[] args) {
        try {
            System.out.println("Tutorial 15");
            System.out.println("----------");

            //Create an instance of the object that generates Excel files, having 2 sheets
            ExcelDocument xls = new ExcelDocument(2);

            //Set the sheet names
            ExcelWorksheet xlsTab1 = (ExcelWorksheet) xls.easy_getSheetAt(0);
            ExcelWorksheet xlsTab2 = (ExcelWorksheet) xls.easy_getSheetAt(1);
            xlsTab1.setSheetName("First tab");
            xlsTab2.setSheetName("Second tab");

            //Create the hyperlink to an URL (1)
            xlsTab1.easy_addHyperlink(EasyXLS.Constants.HyperlinkType.URL,
                                        "http://www.euoutsourcing.com", "Link to URL", "B2:E2");

            //Create the hyperlink to a FILE  (2)
            xlsTab1.easy_addHyperlink(EasyXLS.Constants.HyperlinkType.FILE,
                                        "c:\\myfile.xls", "Link to file", "B3");

            //Create the hyperlink to an UNC (3)
            xlsTab1.easy_addHyperlink(EasyXLS.Constants.HyperlinkType.UNC,
                                        "\\\\computerName\\Folder\\file.txt", "Link to UNC", "B4:D4");

            //Create the hyperlink to a CELL  (4)
            xlsTab1.easy_addHyperlink(EasyXLS.Constants.HyperlinkType.CELL,
                                        "'Second tab'!D3", "Link to CELL", "B5");

            //Creating a name for the second sheet
            xlsTab2.easy_addName("Name", "=Second tab!$A$1:$A$4");

            //Create the hyperlink to a name (5)
            xlsTab1.easy_addHyperlink(EasyXLS.Constants.HyperlinkType.CELL, 
                                        "Name", "Link to a name", "B6");

            //Generate the file
            System.out.println("Writing file: C:\\Samples\\Tutorial15.xls");
            xls.easy_WriteXLSFile("C:\\Samples\\Tutorial15.xls");

            //Confirm generation
            if (xls.easy_getError().equals(""))
                System.out.println("File successfully created.");
            else
                System.out.println("Error encountered: " + xls.easy_getError());

            //Dispose memory
            xls.Dispose();
        }
        catch (Exception ex) {
            ex.printStackTrace();
        }
    }
}
